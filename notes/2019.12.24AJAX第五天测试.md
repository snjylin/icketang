3：简述封装JSONP的过程？
```
function jsonp(url, data, callback) {
    let str = '';
    for (let i in data) {
        str += '&' + i + '=' + data[i];
    }
    str = str.slice(1);
    let script = document.createElement('script');
    script.src = url + '?' + str;
    window[data.callback] = callback;
    document.body.appendChild(script);
    script.onload = function() {
        document.body.removeChild(this);
        window[data.back];
    }
}
```
